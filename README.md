		<ul><li>pip install SPARQLWrapper</li></ul>
	</li>

		<ul>
			<li>pip install numpy</li>
			<li>pip install pandas</li>
		</ul>

	<li>SpaCy:

			<li>pip install -U spacy</li>

			<li>python -m spacy download pt_core_news_sm</li>
		</ul>
	</li>
	<li>SciKit Learn:
		<ul>
			<li>pip install -U scikit-learn</li>
		</ul>
	</li>
